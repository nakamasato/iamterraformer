    * path

    + name (Required)
    ==
    * path
}

    + name  (Optional, Forces new resource)
    ==
    * path
    * permissions_boundary
    * force_destroy
    * tags

    + name  (Optional, Forces new resource)
    ==

    * description

}

}

}
